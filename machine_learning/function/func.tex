\subsection{log loss}

\begin{jcode}{python}{log loss计算}{code:logloss}
import numpy as np
def log_loss(true_y, pred_h):
    return -np.mean(true_y * np.log(pred_h) + (1 - true_y) * np.log(1 - pred_h))
\end{jcode}
